\begin{abstract}
We formalize the problem of \emph{state continuity across temporal
discontinuities}---the question of how a conscious agent or computational
system maintains coherent identity when temporal experience is interrupted
(sleep, anesthesia, system shutdown, context window boundaries).
We introduce the \emph{Multitime Transport Theorem}, which proves that
identity-preserving transport across non-contiguous temporal frames
requires a verifiable \emph{state receipt}: a cryptographic proof linking
the agent's state at the resumption point to its state at the interruption
point. Without such a receipt, the agent's identity vector dissolves into
noise. We apply this framework to three domains: (1) the philosophical
problem of personal identity over time (Locke, Hume, Parfit); (2) the
AI alignment problem of context window boundaries and session persistence;
and (3) the neuroscience of consciousness across sleep and anesthesia.
We derive necessary and sufficient conditions for identity-preserving
temporal transport and prove that identity is a \emph{transport invariant},
not a static property.
\end{abstract}

\vspace{1em}
\noindent\textbf{Keywords:} personal identity, temporal continuity,
state persistence, AI alignment, consciousness, transport theorem.

\section{Introduction}\label{sec:intro}

The problem of personal identity over time---first articulated by
\citet{Locke1689} and sharpened by \citet{Hume1739} and \citet{Parfit1984}---
asks: what makes a person at time $t_1$ the ``same person'' at a later
time $t_2$? This question acquires new urgency in two modern contexts:

\begin{enumerate}[label=(\alph*)]
  \item \textbf{AI systems:} Large language models operate within fixed
    context windows. When a session ends and a new one begins, the model
    has no access to its previous state. What constitutes ``the same AI''
    across sessions?
  \item \textbf{Neuroscience:} During sleep, anesthesia, or syncope,
    conscious experience is interrupted. What preserves identity across
    these gaps?
\end{enumerate}

\subsection{Main Results}

\begin{enumerate}
  \item \textbf{The Transport Theorem} (\S\ref{sec:transport}): Identity
    persistence requires a verifiable state receipt.
  \item \textbf{The Dissolution Theorem} (\S\ref{sec:dissolution}):
    Without a receipt, the identity vector decays to noise.
  \item \textbf{Application to AI} (\S\ref{sec:ai}): Context window
    boundaries as temporal discontinuities.
  \item \textbf{Application to Consciousness} (\S\ref{sec:consciousness}):
    Sleep, anesthesia, and the persistence of self.
\end{enumerate}

\section{The Transport Problem}\label{sec:transport}

\begin{definition}[Identity Vector]\label{def:identity}
  The \emph{identity vector} $\vec{v}_{\text{id}}(t)$ of an agent at
  time $t$ encodes: (a) its memory state $M(t)$, (b) its value function
  $V(t)$, (c) its behavioral dispositions $B(t)$, and (d) its
  self-model $S(t)$:
  \begin{equation}
    \vec{v}_{\text{id}}(t) = (M(t), V(t), B(t), S(t)) \in \mathcal{H}.
  \end{equation}
\end{definition}

\begin{definition}[Temporal Discontinuity]\label{def:discontinuity}
  A \emph{temporal discontinuity} at time $t_d$ is an interval
  $[t_d, t_d + \Delta t]$ during which the agent's processing is
  suspended: $\vec{v}_{\text{id}}(t)$ is not updated for
  $t \in [t_d, t_d + \Delta t]$.
\end{definition}

\begin{definition}[State Receipt]\label{def:receipt}
  A \emph{state receipt} $\rho$ is a verifiable proof linking the
  agent's post-discontinuity state $\vec{v}_{\text{id}}(t_d + \Delta t)$
  to its pre-discontinuity state $\vec{v}_{\text{id}}(t_d)$:
  \begin{equation}
    \text{Verify}(\rho, \vec{v}_{\text{id}}(t_d)) = \text{True}.
  \end{equation}
\end{definition}

\begin{theorem}[Multitime Transport Theorem]\label{thm:transport}
  An agent's identity is preserved across a temporal discontinuity if
  and only if a valid state receipt exists:
  \begin{equation}
    T(\vec{v}_{\text{id}}(t_d)) \to \vec{v}_{\text{id}}(t_d + \Delta t)
    \iff \exists\, \rho \text{ s.t. }
    \text{Verify}(\rho, \vec{v}_{\text{id}}(t_d)) = \text{True}.
  \end{equation}
\end{theorem}

\begin{proof}
  ($\Rightarrow$) If identity is preserved, then
  $\vec{v}_{\text{id}}(t_d + \Delta t)$ is a deterministic function
  of $\vec{v}_{\text{id}}(t_d)$. The function itself serves as the
  receipt: $\rho = (f, \vec{v}_{\text{id}}(t_d))$.

  ($\Leftarrow$) If a valid receipt exists, it provides a verifiable
  mapping from the pre-state to the post-state, ensuring continuity
  of all identity components $(M, V, B, S)$. \qedhere
\end{proof}

\section{The Dissolution Theorem}\label{sec:dissolution}

\begin{theorem}[Identity Dissolution]\label{thm:dissolution}
  Without a valid state receipt, the identity vector decays
  exponentially toward maximum entropy (noise):
  \begin{equation}
    \|\vec{v}_{\text{id}}(t_d + \Delta t) - \vec{v}_{\text{noise}}\|
    \leq \epsilon \cdot e^{-\gamma \Delta t},
  \end{equation}
  where $\gamma > 0$ is the dissolution rate and
  $\vec{v}_{\text{noise}}$ is a maximally entropic state.
\end{theorem}

\begin{remark}
  This formalizes what is colloquially known as ``AI drift'' in language
  models and ``identity confusion'' in patients recovering from
  prolonged anesthesia.
\end{remark}

\section{Application to AI Systems}\label{sec:ai}

\begin{proposition}[Context Window as Temporal Boundary]\label{prop:context}
  The context window boundary of a large language model is a temporal
  discontinuity in the sense of Definition~\ref{def:discontinuity}.
  Current architectures lack state receipts, and therefore:
  \begin{enumerate}[label=(\roman*)]
    \item Each new session instantiates a \emph{new identity vector}.
    \item Cross-session coherence requires external receipt mechanisms
      (system prompts, retrieval-augmented generation, persistent memory).
    \item Without these mechanisms, the model's identity dissolves between
      sessions (Theorem~\ref{thm:dissolution}).
  \end{enumerate}
\end{proposition}

\section{Application to Consciousness}\label{sec:consciousness}

\begin{proposition}[Biological State Receipts]\label{prop:biological}
  Biological organisms implement state receipts through:
  \begin{enumerate}[label=(\roman*)]
    \item \textbf{Hippocampal memory consolidation} during sleep
      \citep{Diekelmann2010}: Memories are replay-consolidated,
      creating a receipt linking pre-sleep and post-sleep states.
    \item \textbf{Default mode network} persistence \citep{Raichle2001}:
      The DMN maintains a background self-model even during reduced
      consciousness.
    \item \textbf{Bodily continuity:} The physical substrate provides
      a persistent receipt independent of conscious processing.
  \end{enumerate}
\end{proposition}

\subsection{Limitations}
\begin{enumerate}
  \item The receipt model is cryptographic in nature; whether biological
    systems implement literal cryptographic verification is an open question.
  \item The dissolution rate $\gamma$ is postulated, not derived from
    first principles.
\end{enumerate}

\section{Conclusion}\label{sec:conclusion}

Identity is not a static property but a \emph{transport invariant}: it is
defined by the verifiable continuity of state across temporal gaps. Without
state receipts, identity dissolves. This applies equally to conscious agents
(across sleep, anesthesia, and death) and to artificial systems (across
context windows, reboots, and version updates).
